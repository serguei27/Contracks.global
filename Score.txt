// ── STATE ─────────────────────────────────────────────────────────────────
let currentScorecard = 0;
let currentQuestion = 0;
let answers = new Array(8).fill(null);
let userName = '';
let userEmail = '';

// ── FUNCTIONS ─────────────────────────────────────────────────────────────

function showScreen(id) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen-' + id).classList.add('active');
  window.scrollTo(0, 0);
}

function startScorecard(index) {
  currentScorecard = index;
  currentQuestion = 0;
  answers = new Array(8).fill(null);
  const sc = scorecards[index];

  // Set header
  const badge = document.getElementById('quiz-badge');
  badge.textContent = sc.speed + ' — ' + sc.name;
  badge.className = 'quiz-speed-badge ' + sc.id;

  document.getElementById('quiz-title').textContent = sc.title;
  document.getElementById('quiz-subtitle').textContent = sc.subtitle;

  // Progress bar class
  document.querySelector('.progress-bar').className = 'progress-bar ' + sc.id;

  // Build questions
  buildQuestions(sc);
  showQuestion(0);
  showScreen('quiz');
}

function buildQuestions(sc) {
  const container = document.getElementById('questions-container');
  container.innerHTML = '';
  sc.questions.forEach((q, qi) => {
    const block = document.createElement('div');
    block.className = 'question-block' + (qi === 0 ? ' active' : '');
    block.id = 'q-' + qi;
    block.innerHTML = `
      <div class="question-num">Question ${qi + 1} of 8</div>
      <div class="question-text">${q}</div>
      <div class="answer-options">
        ${[
          { score: 0, label: 'Not yet — this isn\'t in place' },
          { score: 1, label: 'In progress — partially done' },
          { score: 2, label: 'Done — this is solid' }
        ].map(opt => `
          <button class="answer-btn" onclick="selectAnswer(${qi}, ${opt.score}, event.target)">
            <span class="answer-score">${opt.score}</span>
            <span>${opt.label}</span>
          </button>
        `).join('')}
      </div>
    `;
    container.appendChild(block);
  });
}

function showQuestion(index) {
  document.querySelectorAll('.question-block').forEach(b => b.classList.remove('active'));
  document.getElementById('q-' + index).classList.add('active');

  const pct = ((index + 1) / 8) * 100;
  document.getElementById('progress-fill').style.width = pct + '%';
  document.getElementById('progress-text').textContent = 'Question ' + (index + 1) + ' of 8';

  const btnNext = document.getElementById('btn-next');
  const btnPrev = document.getElementById('btn-prev');

  btnNext.disabled = answers[index] === null;
  btnNext.textContent = index === 7 ? 'See my score →' : 'Continue →';
  btnPrev.style.visibility = index === 0 ? 'hidden' : 'visible';

  // Restore selected state
  if (answers[index] !== null) {
    const btns = document.querySelectorAll('#q-' + index + ' .answer-btn');
    btns.forEach(b => {
      const score = parseInt(b.querySelector('.answer-score').textContent);
      if (score === answers[index]) b.classList.add('selected');
    });
  }
}

function selectAnswer(qi, score, btn) {
  // Walk up to the actual answer-btn element in case tap landed on child
  while (btn && !btn.classList.contains('answer-btn')) {
    btn = btn.parentElement;
  }
  if (!btn) return;
  answers[qi] = score;
  const btns = btn.closest('.answer-options').querySelectorAll('.answer-btn');
  btns.forEach(b => b.classList.remove('selected'));
  btn.classList.add('selected');
  document.getElementById('btn-next').disabled = false;
}

function nextQuestion() {
  if (answers[currentQuestion] === null) return;
  if (currentQuestion < 7) {
    currentQuestion++;
    showQuestion(currentQuestion);
  } else {
    // Go to email screen
    const total = answers.reduce((a, b) => a + b, 0);
    document.getElementById('score-blur').textContent = total + '/16';
    showScreen('email');
  }
}

function prevQuestion() {
  if (currentQuestion > 0) {
    currentQuestion--;
    showQuestion(currentQuestion);
  }
}

function submitEmail(e) {
  e.preventDefault();
  userName = document.getElementById('email-name').value;
  userEmail = document.getElementById('email-address').value;
  const company = document.getElementById('email-company').value;

  // Send email notification (mailto fallback)
  const sc = scorecards[currentScorecard];
  const total = answers.reduce((a, b) => a + b, 0);
  const tier = sc.tiers.find(t => total >= t.range[0] && total <= t.range[1]);

  // Build mailto
  const subject = encodeURIComponent('Contracks Scorecard — ' + sc.name + ' — ' + userName);
  const body = encodeURIComponent(
    'Name: ' + userName + '\n' +
    'Email: ' + userEmail + '\n' +
    'Company: ' + (company || 'Not provided') + '\n' +
    'Scorecard: ' + sc.speed + ' — ' + sc.name + '\n' +
    'Score: ' + total + '/16\n' +
    'Tier: ' + tier.label + '\n\n' +
    'Answers:\n' +
    sc.questions.map((q, i) => (i+1) + '. ' + q + ' — ' + answers[i] + '/2').join('\n')
  );

  // Open mailto in background
  const link = document.createElement('a');
  link.href = 'mailto:spoppeleer@contracks.global?subject=' + subject + '&body=' + body;
  link.style.display = 'none';
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);

  showResults();
}

function showResults() {
  const sc = scorecards[currentScorecard];
  const total = answers.reduce((a, b) => a + b, 0);
  const tier = sc.tiers.find(t => total >= t.range[0] && total <= t.range[1]);

  // Score circle
  document.getElementById('result-score-num').textContent = total;

  // Tier badge
  const badge = document.getElementById('result-tier-badge');
  badge.textContent = tier.label;
  badge.style.background = tier.color + '22';
  badge.style.color = tier.color;
  badge.style.border = '1px solid ' + tier.color + '55';
  badge.style.borderRadius = '20px';
  badge.style.padding = '8px 20px';
  badge.style.fontSize = '0.8rem';
  badge.style.fontWeight = '700';
  badge.style.letterSpacing = '0.1em';
  badge.style.textTransform = 'uppercase';

  // Score circle border
  document.querySelector('.result-score-display').style.borderColor = tier.color;
  document.querySelector('.result-score-num').style.color = tier.color;

  document.getElementById('result-heading').textContent = tier.heading;
  document.getElementById('result-desc').textContent = tier.desc;

  // Breakdown
  const breakdownList = document.getElementById('breakdown-list');
  breakdownList.innerHTML = sc.questions.map((q, i) => `
    <div class="breakdown-item">
      <div class="breakdown-q">${q}</div>
      <div class="breakdown-score score-${answers[i]}">${answers[i] === 2 ? '✓ Done' : answers[i] === 1 ? '~ Progress' : '✗ Not yet'}</div>
    </div>
  `).join('');

  // CTA
  document.getElementById('result-cta-title').textContent = tier.ctaTitle;
  document.getElementById('result-cta-desc').textContent = tier.ctaDesc;
  const ctaBtn = document.getElementById('result-cta-btn');
  ctaBtn.textContent = tier.ctaBtn;
  ctaBtn.href = 'mailto:spoppeleer@contracks.global?subject=' + encodeURIComponent('Discovery call — ' + userName + ' — ' + sc.name + ' ' + total + '/16');

  showScreen('results');
}

  showScreen('landing');
